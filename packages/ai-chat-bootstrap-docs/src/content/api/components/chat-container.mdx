# ChatContainer

Main chat interface component with message display and input.

```tsx
interface ChatContainerProps {

  messages: UIMessage[];
  input: string;
  onInputChange: (e: ChangeEvent<HTMLInputElement | HTMLTextAreaElement>) => void;
  onSubmit: (e?: FormEvent) => void;
  isLoading?: boolean;
  error?: Error | null;
  placeholder?: string;
  className?: string;
}
```
